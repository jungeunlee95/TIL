int num = 4;
num = num + 10;  //4+10= 임의의 상수값을 num이 갖도록 즉, 자신의 수에 10을 더한 값을 갖게 한다
                //우항을 좌항에 대입, 더한 다음에 업데이트 해라 
num += 10;       //위의 식과 같은 의미
                 //무엇을 사용하던 상관없지만, 다른 코드를 봤을 때 의미를 알아야 되기 때문에 다 알아야함.
//프로그램 사용할 때, 1씩 증가하거나 1씩 감소하는 경우가 많음

num = num+1;
num += 1;    //둘중하나 사용해야함- but 편하게 코드하기 위하여

++num;       //전위 증가 연산자 - 선행처리
num++;       //후위             - 후행처리
 
/* 연산자 종류
   단항: 어떤 변수의 값에 -,+를 지정할때 - 단항연산자
   이항: 일반적으로는, 두개의 변수간의 관계를 지정하는 것-이항연산자 삼항,4항은 컴퓨터에선 안그럼, 안똑똑하기때문에, 무조건 두개씩 연산
                                                         A+B+C = (A+B)+C = ((A+B)+C) 이런식을 두개씩 묶어서 연산함.
   내 변수에 1씩 증가시키고 싶은데 궅이 이항을 써야돼? -> 단항을 쓰고싶어
*/
 
//혼자만 쓸때는 차이가 없지만, 다른 문장과 결합을 할 때 차이가 생김
++num;
num++;

ex) 
num = 10;
System.out.println(++num);  //num을 하나 증가 시킨 후에, print(num)를 찍어라 -> 11

num = 10;
System.out.println(num++);  //num을 먼저 print한 후에 증가해라. ->10으로 출력이 된 후, ;(세미콜론)이후의 나중의 num값이 11로 증가가 되는거임
